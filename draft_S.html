data['Average_Completion_Rate'] = data[['Average_Completion_Rate_Male', 'Average_Completion_Rate_Female']].mean(axis=1)

# Select relevant columns for the radar chart
radar_data = data[['Country', 'Average_OOSR', 'Average_Completion_Rate',
                   'Lower_Secondary_End_Proficiency_Reading', 'Lower_Secondary_End_Proficiency_Math',
                   'Ladder score', 'Logged GDP per capita', 'Social support',
                   'Healthy life expectancy', 'Freedom to make life choices']]

# Normalize the data for radar chart
scaler = MinMaxScaler()
radar_data_normalized = radar_data.copy()
radar_data_normalized.iloc[:, 1:] = scaler.fit_transform(radar_data.iloc[:, 1:])

# Function to create radar chart
def create_radar_chart(country):
    labels = radar_data.columns[1:]
    num_vars = len(labels)
    
    # Create a 360 degree circle
    angles = np.linspace(0, 2 * np.pi, num_vars, endpoint=False).tolist()
    
    # The radar chart is circular, so we need to "complete the loop" and append the start to the end.
    angles += angles[:1]
    
    # Select the normalized data for the chosen country
    values_normalized = radar_data_normalized[radar_data_normalized['Country'] == country].iloc[0, 1:].tolist()
    values_normalized += values_normalized[:1]
    
    # Select the actual data values for the chosen country for reference
    values_actual = radar_data[radar_data['Country'] == country].iloc[0, 1:].tolist()
    
    fig, ax = plt.subplots(figsize=(8, 8), subplot_kw=dict(polar=True))
    
    ax.fill(angles, values_normalized, color='skyblue', alpha=0.4)
    ax.plot(angles, values_normalized, color='dodgerblue', linewidth=2)
    
    ax.set_yticklabels([])
    ax.set_xticks(angles[:-1])
    ax.set_xticklabels(labels, fontsize=12)
    
    for i, label in enumerate(labels):
        angle_rad = angles[i]
        ha = "right"
        if angle_rad < np.pi / 2 or angle_rad > 3 * np.pi / 2:
            ha = "left"
        ax.text(angle_rad, values_normalized[i], f"{values_actual[i]:.2f}", size=10, horizontalalignment=ha, verticalalignment="center", color="black")
    
    ax.set_title(f'Radar Chart for {country}', size=18, color='dodgerblue', y=1.1)
    
    plt.subplots_adjust(top=0.85, bottom=0.15)
    plt.show()

# Dropdown menu to select country
countries = radar_data['Country'].unique()

country_dropdown = widgets.Dropdown(

    values_normalized_1 += values_normalized_1[:1]

    values_normalized_2 += values_normalized_2[:1]

    values_actual_2 = radar_data[radar_data['Country'] == country2].iloc[0, 1:].tolist()
